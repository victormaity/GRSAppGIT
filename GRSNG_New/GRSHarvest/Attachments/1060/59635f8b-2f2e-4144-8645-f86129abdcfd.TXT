          SELECT  distinct category_name FROM(SELECT tr.trial_id,NVL(tf.category_code,oa.category_code) AS category_code         ,NVL(tf.category_name,oa.category_name) AS category_name FROM ci_public_staging.jp_trial tr LEFT JOIN (         SELECT tf.trial_id,ocl.category_code,ocl.category_name FROM ci_public_staging.jp_trial_funder tf         JOIN ci_public_staging.ng_org_category_ft_mapping ocf ON ocf.funder_type_id = tf.funder_type_id         JOIN ci_public_staging.ng_organization_category_list ocl ON ocl.category_code = ocf.category_id         ) tf ON tf.trial_id = tr.trial_id LEFT JOIN (         SELECT ct.trial_id,ncc.category_code,ncc.category_name FROM ci_public_staging.jp_company_trial ct         JOIN ci_public_staging.ng_company_category ncc ON ncc.company_id = ct.company_id WHERE category_code IN ('CMP','ACA','GOV','NGO','OTH')         ) oa ON oa.trial_id = tr.trial_id) WHERE category_code IS NOT NULL and trial_id in '98587'
